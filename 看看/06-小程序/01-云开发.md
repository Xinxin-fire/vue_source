### 云开发

#### 1.初始化云函数

```
App({
  onLaunch: function () {
    if (!wx.cloud) {
      console.error('请使用 2.2.3 或以上的基础库以使用云能力')
    } else {
      wx.cloud.init({
        // env 参数说明：
        //   env 参数决定接下来小程序发起的云开发调用（wx.cloud.xxx）会默认请求到哪个云环境的资源
        //   此处请填入环境 ID, 环境 ID 可打开云控制台查看
        //   如不填则使用默认环境（第一个创建的环境）
        env: 'frontend-knowledge-3dcyb9908ddf3',
        traceUser: true,
      })
    }

    this.globalData = {}
  }
})
```

#### 2.查询数据

```
// 1. 获取数据库引用
    const db = wx.cloud.database();
    // 2. 构造查询语句
    // collection 方法获取一个集合的引用
    // where 方法传入一个对象，数据库返回集合中字段等于指定值的 JSON 文档。API 也支持高级的查询条件（比如大于、小于、in 等），具体见文档查看支持列表
    // get 方法会触发网络请求，往数据库取数据
    db.collection('titles').where({}).get({
    }).then(res => {
      console.log(res)
    }).catch(err => {
      console.log(err)
    });
```

#### 3.添加数据

```
// 1. 获取数据库引用
    const db = wx.cloud.database();
    // 2. 构造查询语句
    // collection 方法获取一个集合的引用
    db.collection('titles').add({
       //添加数据
       data: {
           name: '',
           value: ''
       }
    }).then(res => {
      console.log(res)
    }).catch(err => {
      console.log(err)
    });
```

#### 6.初始化云函数

- 新建云函数的文件夹
- 在project.config.json中配置："cloudfunctionRoot": "cloudfunctions/"

#### 7.新建云函数

在云函数的文件夹下新建node.js函数

在云函数中调用对应集合的get方法，并部署云函数

```
// 云函数入口文件
const cloud = require('wx-server-sdk')

cloud.init({
  env: cloud.DYNAMIC_CURRENT_ENV
})

// 云函数入口函数
exports.main = async (event, context) => {
  return cloud.database().collection('getHtmlTitles').get();
}
```

#### 8.在本地小程序中调用云函数

```
// 云函数的引用
    wx.cloud.callFunction({
      name: 'getHtmlTitles'
    }).then(res => {
      console.log(res)
    })
```
